7/// [CommentLine](crate::ast::CommentLine) ::= ("###" | "##" | "#") ("\u0020" comment_char*)? line_end
8///
9/// Adjacent comment lines of the same comment type are joined together during
10/// the AST construction.
11#[derive(Clone, Debug, PartialEq)]
12#[cfg_attr(feature = "hash", derive(Eq, PartialOrd, Ord, Hash))]
13#[cfg_attr(feature = "serde", derive(Deserialize, Serialize))]
14pub struct CommentLine {
15    lead: String,
16    comment: Option<String>,
17}
18
19impl CommentLine {
20    /// Constructs a new `CommentLine` representing a comment line in a Fluent Translation List (FTL) file.
21    ///
22    /// # Arguments
23    /// * `lead` - The leading part of the line, including the `#` symbols and any immediate following whitespace.
24    /// * `comment` - The optional comment text after the initial whitespace following the `#` symbols.
25    ///   If `None`, the line consists only of the leading hashes (an "empty" comment).
26    pub fn new(lead: String, comment: Option<String>) -> Self {
27        Self { lead, comment }
28    }
29}
30
31#[cfg(feature = "walker")]
32impl Walkable for CommentLine {
33    fn walk(&self, visitor: &mut dyn Visitor) {
34        visitor.visit_comment_line(self);
35    }
36}
37
38impl std::fmt::Display for CommentLine {
39    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
40        let comment = self.comment.as_ref().map_or("".into(), |c| format!(" {c}"));
41        writeln!(f, "{}{}", self.lead, comment)
42    }
43}
