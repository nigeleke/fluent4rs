9/// [Entry](crate::ast::Entry) ::= ([Message](crate::ast::Message) line_end)
10///  | ([Term](crate::ast::Term) line_end)
11///  | [CommentLine](crate::ast::CommentLine)
12///
13/// Entries are the main building blocks of Fluent.
14///
15/// They define translations and contextual and semantic information about the
16/// translations. During the AST construction, adjacent comment lines of the same
17/// comment type (defined by the number of #) are joined together. Single-# comments
18/// directly preceding [Message](crate::ast::Message)s and [Term](crate::ast::Term)s
19/// are attached to the [Message](crate::ast::Message) or [Term](crate::ast::Term)
20/// and are not standalone Entries.
21#[derive(Clone, Debug, PartialEq)]
22#[cfg_attr(feature = "hash", derive(Eq, PartialOrd, Ord, Hash))]
23#[cfg_attr(feature = "serde", derive(Deserialize, Serialize))]
24pub enum Entry {
25    #[doc(hidden)]
26    Message(Message),
27
28    #[doc(hidden)]
29    Term(Term),
30
31    #[doc(hidden)]
32    CommentLine(CommentLine),
33}
34
35#[cfg(feature = "walker")]
36impl Walkable for Entry {
37    fn walk(&self, visitor: &mut dyn Visitor) {
38        visitor.visit_entry(self);
39        match self {
40            Self::Message(message) => Walker::walk(message, visitor),
41            Self::Term(term) => Walker::walk(term, visitor),
42            Self::CommentLine(comment) => Walker::walk(comment, visitor),
43        }
44    }
45}
46
47impl std::fmt::Display for Entry {
48    fn fmt(&self, f: &mut std::fmt::Formatter<'_>) -> std::fmt::Result {
49        let value = match self {
50            Self::Message(message) => message.to_string(),
51            Self::Term(term) => term.to_string(),
52            Self::CommentLine(line) => line.to_string(),
53        };
54        write!(f, "{value}")
55    }
56}
