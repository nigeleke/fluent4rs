1use thiserror::Error;

3#[derive(Debug, Error)]
4/// Represents errors that can occur during parsing of a Fluent Translation List (FTL) resource.
5pub enum Fluent4rsError {
6    /// Error raised by pom parser duing parsing of fluent file text.
7    #[error(transparent)]
8    PomParser(#[from] pom::Error),
9
10    /// Parsing completed successully according to the fluent grammar, but Junk entries
11    /// found in resultant AST. By default these are treated as an error in fluent4rs.
12    /// Use `parse_with_junk` to include these AST elements in the final tree.
13    #[error("Unwanted junk found in Fluent grammar: {0}")]
14    UnwantedJunk(String),
15}
