1use thiserror::Error;
2
3#[derive(Debug, Error)]
4/// Represents errors that can occur during parsing of a Fluent Translation List (FTL) resource.
5pub enum Fluent4rsError {
6    /// Error raised by pom parser duing parsing of fluent file text.
7    #[cfg(feature = "parser-pom")]
8    #[error(transparent)]
9    PomParser(#[from] pom::Error),
10
11    /// Error raised by chumsky parser duing parsing of fluent file text.
12    #[cfg(feature = "parser-chumsky")]
13    #[error("{0}")]
14    ChumskyParser(String),
15
16    /// Parsing completed successully according to the fluent grammar, but Junk entries
17    /// found in resultant AST. By default these are treated as an error in fluent4rs.
18    /// Use `parse_with_junk` to include these AST elements in the final tree.
19    #[error("Unwanted junk found in Fluent grammar: {0}")]
20    UnwantedJunk(String),
21}
